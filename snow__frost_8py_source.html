37  def initial(self):
38  """ initial part of the snow and frost module
39  """
40  self.var.TotalPrecipitation = globals.inZero.copy()
41  self.var.DeltaTSnow = 0.9674 * loadmap('ElevationStD') * loadmap('TemperatureLapseRate')
42 
43  # Difference between (average) air temperature at average elevation of
44  # pixel and centers of upper- and lower elevation zones [deg C]
45  # ElevationStD: Standard Deviation of the DEM from Bodis (2009)
46  # 0.9674: Quantile of the normal distribution: u(0,833)=0.9674
47  # to split the pixel in 3 equal parts.
48  self.var.SnowDayDegrees = 0.9856
49  # day of the year to degrees: 360/365.25 = 0.9856
50 
51  self.var.IceDayDegrees = 1.915
52  # days of summer (15th June-15th Sept.) to degree: 180/(259-165)
53  self.var.SnowSeason = loadmap('SnowSeasonAdj') * 0.5
54  # default value of range of seasonal melt factor is set to 1
55  # 0.5 x range of sinus function [-1,1]
56  self.var.TempSnow = loadmap('TempSnow')
57  self.var.SnowFactor = loadmap('SnowFactor')
58  self.var.SnowMeltCoef = loadmap('SnowMeltCoef')
59  self.var.TempMelt = loadmap('TempMelt')
60 
61  SnowCoverAInit = loadmap('SnowCoverAInitValue')
62  SnowCoverBInit = loadmap('SnowCoverBInitValue')
63  SnowCoverCInit = loadmap('SnowCoverCInitValue')
64  self.var.SnowCoverS = [SnowCoverAInit, SnowCoverBInit, SnowCoverCInit]
65 
66  # initial snow depth in elevation zones A, B, and C, respectively [mm]
67  self.var.SnowCoverInit = (SnowCoverAInit + SnowCoverBInit + SnowCoverCInit) / 3
68  # Pixel-average initial snow cover: average of values in 3 elevation
69  # zones

87  def dynamic(self):
88  """ dynamic part of the snow module
89  """
90 
91  SeasSnowMeltCoef = self.var.SnowSeason * np.sin(math.radians((self.var.CalendarDay - 81)
92  * self.var.SnowDayDegrees)) + self.var.SnowMeltCoef
93 
94  # SeasSnowMeltCoef = SnowSeason * sin((CalendarDay-81)* SnowDayDegrees) + SnowMeltCoef;
95 
96  # sinus shaped function between the
97  # annual minimum (December 21st) and annual maximum (June 21st)
98  # SummerSeason = ifthenelse(self.var.CalendarDay > 165,np.sin((self.var.CalendarDay-165)* self.var.IceDayDegrees ),scalar(0.0))
99  # SummerSeason = ifthenelse(self.var.CalendarDay > 259,0.0,SummerSeason)
100  if (self.var.CalendarDay > 165) and (self.var.CalendarDay < 260):
101  SummerSeason = np.sin(math.radians((self.var.CalendarDay - 165) * self.var.IceDayDegrees))
102  else:
103  SummerSeason = 0.0
104 
105  self.var.Snow = globals.inZero.copy()
106  self.var.Rain = globals.inZero.copy()
107  self.var.SnowMelt = globals.inZero.copy()
108  self.var.SnowCover = globals.inZero.copy()
109 
110  for i in xrange(3):
111  TavgS = self.var.Tavg + self.var.DeltaTSnow * (i - 1)
112  # Temperature at center of each zone (temperature at zone B equals Tavg)
113  # i=0 -> highest zone
114  # i=2 -> lower zone
115  SnowS = np.where(TavgS < self.var.TempSnow, self.var.SnowFactor * self.var.Precipitation, globals.inZero)
116  # Precipitation is assumed to be snow if daily average temperature is below TempSnow
117  # Snow is multiplied by correction factor to account for undercatch of
118  # snow precipitation (which is common)
119  RainS = np.where(TavgS >= self.var.TempSnow, self.var.Precipitation, globals.inZero)
120  # if it's snowing then no rain
121  SnowMeltS = (TavgS - self.var.TempMelt) * SeasSnowMeltCoef * (1 + 0.01 * RainS) * self.var.DtDay
122 
123  if i < 2:
124  IceMeltS = self.var.Tavg * 7.0 * self.var.DtDay * SummerSeason
125  # if i = 0 and 1 -> higher and middle zone
126  else:
127  IceMeltS = TavgS * 7.0 * self.var.DtDay * SummerSeason
128 
129  SnowMeltS = np.maximum(np.minimum(SnowMeltS + IceMeltS, self.var.SnowCoverS[i]), globals.inZero)
130  self.var.SnowCoverS[i] = self.var.SnowCoverS[i] + SnowS - SnowMeltS
131  self.var.Snow += SnowS
132  self.var.Rain += RainS
133  self.var.SnowMelt += SnowMeltS
134  self.var.SnowCover += self.var.SnowCoverS[i]
135 
136  self.var.Snow /= 3
137  self.var.Rain /= 3
138  self.var.SnowMelt /= 3
139  self.var.SnowCover /= 3
140  # all in pixel [mm]

149  FrostIndexChangeRate = -(1 - self.var.Afrost) * self.var.FrostIndex - self.var.Tavg * \
150  np.exp(-0.04 * self.var.Kfrost * self.var.SnowCover / self.var.SnowWaterEquivalent)
151  # FrostIndexChangeRate=self.var.AfrostIndex - self.var.Tavg* pcraster.exp(self.var.Kfrost*self.var.SnowCover*self.var.InvSnowWaterEquivalent)
152  # Rate of change of frost index (expressed as rate, [degree days/day])
153  self.var.FrostIndex = np.maximum(self.var.FrostIndex + FrostIndexChangeRate * self.var.DtDay, 0)
154  # frost index in soil [degree days] based on Molnau and Bissel (1983, A Continuous Frozen Ground Index for Flood
155  # Forecasting. In: Maidment, Handbook of Hydrology, p. 7.28, 7.55)
156  # if Tavg is above zero, FrostIndex will stay 0
157  # if Tavg is negative, FrostIndex will increase with 1 per degree C per day
158  # Exponent of 0.04 (instead of 0.4 in HoH): conversion [cm] to [mm]!
159  # Division by SnowDensity because SnowDepth is expressed as equivalent water depth(always less than depth of snow pack)
160  # SnowWaterEquivalent taken as 0.100 (based on density of 100 kg/m3) (Handbook of Hydrology, p. 7.5)
161  # Afrost, (daily decay coefficient) is taken as 0.97 (Handbook of Hydrology, p. 7.28)
162  # Kfrost, (snow depth reduction coefficient) is taken as 0.57 [1/cm], (HH, p. 7.28)
