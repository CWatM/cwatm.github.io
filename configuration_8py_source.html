33  def get(self, section, option, raw=False, vars=None):

42  re_old1 = re.findall('\$\(\w*:\w*\)', r_opt)
43  re_old2 = re.findall('\$\(\w*\)', r_opt)
44 
45  m_new1 = re.findall(re_newintp1, r_opt)

47 
48  if m_new1:

53  sub = self.get(f_section,f_option, vars=vars)

56  else:
