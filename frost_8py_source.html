61  """ dynamic part of the frost index module

63  # FrostIndexChangeRate=-(1-Afrost)*FrostIndex - Tavg*exp(-0.04*Kfrost*SnowCover/SnowWaterEquivalent);
64 
65  FrostIndexChangeRate = -(1 - self.var.Afrost) * self.var.FrostIndex - self.var.Tavg * \
66  np.exp(-0.04 * self.var.Kfrost * self.var.SnowCover / self.var.SnowWaterEquivalent)
67  # FrostIndexChangeRate=self.var.AfrostIndex - self.var.Tavg* pcraster.exp(self.var.Kfrost*self.var.SnowCover*self.var.InvSnowWaterEquivalent)
68  # Rate of change of frost index (expressed as rate, [degree days/day])
69  # CHANGED 9 September 2004:
70  # - first term should be negative
71  # - second term should be subtracted, not added!!
72 
73  self.var.FrostIndex = np.maximum(self.var.FrostIndex + FrostIndexChangeRate * self.var.DtDay, 0)
74  # frost index in soil [degree days]
75  # based on Molnau and Bissel (1983, A Continuous Frozen Ground Index for Flood
76  # Forecasting. In: Maidment, Handbook of Hydrology, p. 7.28, 7.55)
77  # if Tavg is above zero, FrostIndex will stay 0
78  # if Tavg is negative, FrostIndex will increase with 1 per degree C per day
79  # Exponent of 0.04 (instead of 0.4 in HoH): conversion [cm] to [mm]!
80  # Division by SnowDensity because SnowDepth is expressed as equivalent water
81  # depth(always less than depth of snow pack)
82  # SnowWaterEquivalent taken as 0.100 (based on density of 100 kg/m3) (Handbook of Hydrology, p. 7.5)
83  # Afrost, (daily decay coefficient) is taken as 0.97 (Handbook of Hydrology,
84  # p. 7.28)
85  # Kfrost, (snow depth reduction coefficient) is taken as 0.57 [1/cm],
86  # (HH, p. 7.28)
