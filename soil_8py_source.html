213  def improvedArnoScheme(No,kUnsatLow030150,
214  iniWaterStorage, inputNetLqWaterToSoil, directRunoffReduction="Default"):
215  # improvedArnoScheme(No, iniWaterStorage=self.var.soilWaterStorage[No], inputNetLqWaterToSoil=self.var.topWaterLayer[No])
216  # arnoBeta = BCF = b coefficient of soil water storage capacity distribution
217  #
218  # WMIN = root zone water storage capacity, minimum values
219  # WMAX = root zone water storage capacity, area-averaged values
220  # W = actual water storage in root zone
221  # WRANGE = WMAX - WMIN
222  # DW = WMAX-W
223  # WFRAC = DW/WRANGE ; WFRAC capped at 1
224  # WFRACB = DW/WRANGE raised to the power (1/(b+1))
225  # SATFRAC = fractional saturated area
226  # WACT = actual water storage within rootzone
227 
228  Pn = iniWaterStorage +inputNetLqWaterToSoil # Pn = W[TYPE]+Pn;
229  Pn = Pn - np.maximum(self.var.rootZoneWaterStorageMin[No], iniWaterStorage) # Pn = Pn-max(WMIN[TYPE],W[TYPE]);
230 
231  soilWaterStorage = np.where(Pn < 0., self.var.rootZoneWaterStorageMin[No] + Pn, np.maximum(iniWaterStorage, self.var.rootZoneWaterStorageMin[No]))
232  # W[TYPE]= if(Pn<0,WMIN[TYPE]+Pn,max(W[TYPE],WMIN[TYPE]));
233 
234  Pn = np.maximum(0., Pn) # Pn = max(0,Pn);
235 
236  DW = np.maximum(0., self.var.rootZoneWaterStorageCap - soilWaterStorage) # DW = max(0,WMAX[TYPE]-W[TYPE]);
237 
238  WFRAC = np.minimum(1.0, DW / self.var.rootZoneWaterStorageRange[No]) # WFRAC = min(1,DW/WRANGE[TYPE]);
239  WFRACB = WFRAC ** (1. / (1. + self.var.arnoBeta[No])) # WFRACB = WFRAC**(1/(1+BCF[TYPE]));
240 
241  satAreaFrac = np.where(WFRACB > 0., 1. - WFRACB ** self.var.arnoBeta[No], 0.) # SATFRAC_L = if(WFRACB>0,1-WFRACB**BCF[TYPE],0);
242  satAreaFrac = np.maximum(np.minimum(satAreaFrac, 1.0),0.0)
243 
244  actualW = (self.var.arnoBeta[No] + 1) * self.var.rootZoneWaterStorageCap - \
245  self.var.arnoBeta[No] * self.var.rootZoneWaterStorageMin[No] - \
246  (self.var.arnoBeta[No] + 1) * self.var.rootZoneWaterStorageRange[No] * WFRACB
247  # WACT_L = (BCF[TYPE]+1)*WMAX[TYPE]- BCF[TYPE]*WMIN[TYPE]- (BCF[TYPE]+1)*WRANGE[TYPE]*WFRACB;
248 
249 
250  if directRunoffReduction == "Default":
251  directRunoffReduction = np.minimum(kUnsatLow030150, np.sqrt(self.var.kUnsatAtFieldCapLow030150 * kUnsatLow030150))
252  # In order to maintain full saturation and
253  # continuous groundwater recharge/percolation,
254  # the amount of directRunoff may be reduced.
255  # In this case, this reduction is estimated
256  # based on (for two layer case) percLow = np.minimum(KUnSatLow, pcr.sqrt(parameters.KUnSatFC2*KUnSatLow))
257 
258  # directRunoff
259  condition = (self.var.arnoBeta[No] + 1.) * self.var.rootZoneWaterStorageRange[No] * WFRACB
260 
261  """
262  h1= Pn - (self.var.rootZoneWaterStorageCap + directRunoffReduction - soilWaterStorage)
263  h2 = self.var.rootZoneWaterStorageRange[No] * (WFRACB - Pn / ((self.var.arnoBeta[No] + 1.) * self.var.rootZoneWaterStorageRange[No]))
264 
265  #self.rootZoneWaterStorageRange * (self.WFRACB - Pn / ((self.arnoBeta + 1.) * self.rootZoneWaterStorageRange))
266 
267  h3 = (self.var.arnoBeta[No] + 1.)
268  #report(decompress(self.var.arnoBeta[No]), "C:\work\output\h2.map")
269  h4 = h2 ** h3
270  h5 = np.where(Pn >= condition, 0.0, h4)
271  """
272 
273  directRunoff = np.maximum(0.0, Pn - (self.var.rootZoneWaterStorageCap + directRunoffReduction - soilWaterStorage) + \
274  np.where(Pn >= condition, 0.0, \
275  self.var.rootZoneWaterStorageRange[No] * (WFRACB - \
276  Pn / ((self.var.arnoBeta[No] + 1.) * self.var.rootZoneWaterStorageRange[No])) ** (self.var.arnoBeta[No] + 1.)))
277  # Q1_L[TYPE]= max(0,Pn-(WMAX[TYPE]+P2_L[TYPE]-W[TYPE])+
278  # if(Pn>=(BCF[TYPE]+1)*WRANGE[TYPE]*WFRACB, 0,
279  # WRANGE[TYPE]*(WFRACB-Pn/((BCF[TYPE]+1)*WRANGE[TYPE]))**(BCF[TYPE]+1))); #*
280  return directRunoff, WFRACB, satAreaFrac
