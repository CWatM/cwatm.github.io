36  # ************************************************************
37  # ***** INTERCEPTION *****************************************
38  # ************************************************************
39  # Domain: whole pixel (permeable + direct runoff areas)
40  #
41  np.seterr(invalid='ignore',divide='ignore')
42 
43  SMax = np.where(self.var.LAI[sLoop] > 0.1, 0.935 + 0.498 * self.var.LAI[
44  sLoop] - 0.00575 * np.square(self.var.LAI[sLoop]), globals.inZero)
45  SMax = np.where(self.var.LAI[sLoop] > 43.3, 11.718, SMax)
46  # maximum interception [mm]
47  # Van Hoyningen-Huene (1981), p.46
48  # small LAI: no interception
49  # Note that for LAI = 43.3, SMax is at its maximum value of 11.718, and dropping
50  # after that(but LAI should never be that high))
51 
52  self.var.Interception[sLoop] = np.where(SMax > 0.0, np.minimum(SMax - self.var.CumInterception[
53  sLoop], SMax * (1 - np.exp(-0.046 * self.var.LAI[sLoop] * self.var.Rain / SMax))), globals.inZero)
54  self.var.Interception[sLoop] = np.minimum(self.var.Interception[sLoop], self.var.Rain)
55  # Interception (in [mm] per time step)
56  # Smax is calculated from LAI as a constant map (above)
57  # according to Aston (1970), based on Merriam (1960/1970)
58  # 0.046*LAI = k = (1-p) p = 1-0.046*LAI Aston (1970)
59  # LAI must be a pixel average, not the average LAI for PER only!
60 
61  self.var.CumInterception[sLoop] += self.var.Interception[sLoop]
62  # total interception in [mm]

65  # ***** EVAPORATION OF INTERCEPTED WATER *********************
66  # ************************************************************
67  # Domain: whole pixel (permeable + direct runoff areas)

69  TaInterceptionMax = self.var.EWRef * (1 - self.var.LAITerm[sLoop])
70  # Maximum evaporation rate of intercepted water, [mm] per timestep
71  # TaInterception at rate of open water surface only evaporation
72  # of intercepted water in vegetated fraction,hence mutiplication
73  # by(1-LAITerm)
74  self.var.TaInterception[sLoop] = np.maximum(np.minimum(self.var.CumInterception[sLoop], TaInterceptionMax), globals.inZero)

76  # assumption: at first all interception water is evaporated rate is equal to
77  # TaInterceptionMax
78  self.var.CumInterception[sLoop] = np.maximum(self.var.CumInterception[sLoop] - self.var.TaInterception[sLoop], globals.inZero)
79  # evaporated water is subtracted from Cumulative Interception
80  self.var.LeafDrainage[sLoop] = self.var.LeafDrainageK * self.var.CumInterception[sLoop]

83  # on the soil surface
84  self.var.CumInterception[sLoop] = np.maximum(self.var.CumInterception[sLoop] - self.var.LeafDrainage[sLoop], globals.inZero)

133  np.minimum(self.var.RWS[sLoop], 1.0), globals.inZero)
